	\begin{itemize}
		\item When you create a string using the \textbf{new} keyword, it is stored in the heap.
		\item Eg:
		\bigskip
		\codeblock{
			String s1 = new String("lava"); \\
			String s2 = new String("lava"); \\
			String s3 = "lava"; \\
			String s4 = "lava";
		}

	\end{itemize}
