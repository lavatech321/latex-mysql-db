
\begin{flushleft}
	\textbf{Indexing:}
	\begin{itemize}
		\item Strings are sequences of characters.
		\item Each character in a string has an index starting from 0. 
		\newimage{0.3}{content/chapter13/images/index.png}
	\end{itemize}	

	\textbf{Slicing:}
	\begin{itemize}
		\item You can perform string slicing or substring extraction using the index:
		
		\syntaxblock{
		[startIndex,endIndex] => [startIndex,endIndex-1]
		}
		\newimage{0.3}{content/chapter13/images/index2.png}
		
	\end{itemize}

\end{flushleft}
