	\begin{itemize}
		\item Use continue statement inside loops to skip current iteration and continue for the next iteration.
		\bigskip
		\codeblock{
			for(int i=0; i<10; i++) \{ \\
			\s	if(i \% 2 == 0) \\
			\s \s		continue; \\
			\s	System.out.print(i); // Output: 13579 \\
			\}
		}
		
	\end{itemize}

\end{flushleft}
