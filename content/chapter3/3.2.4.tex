
\begin{flushleft}
	
	\begin{itemize}
		\item DELETE query remove specific rows from a table:
		\syntaxblock{
			DELETE FROM table\_name WHERE conditions;
		}
		
		\item DELETE query remove all rows from a table:
		\syntaxblock{
			DELETE FROM table\_name;
		}
		Eg:
		\commandblock{
			DELETE FROM product;	 \\
			DELETE FROM product WHERE id=101;
		}
	\end{itemize}	
	
\end{flushleft}
