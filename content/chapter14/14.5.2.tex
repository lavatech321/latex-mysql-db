	\begin{itemize}
		\item LinkedHashMap is child class of HashMap class. 
		\item It is exactly same as HashMap, except:		
		\tabletwo{
			\textbf{HashMap} & \textbf{LinkedHashMap} \\
			\hline
			Data structure used is \textbf{HashTable} & Data structure is a combination of \textbf{LinkedList + HashTable} \\
			\hline
			Insertion \textbf{order not preserved} \& based on keys & Insertion \textbf{order  preserved} \\
			\hline
			Introucted in 1.2 version & Introduced in 1.4 version \\
		}
	\end{itemize}
