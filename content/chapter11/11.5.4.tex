\setlength{\columnsep}{3pt}
\begin{flushleft}
	
	\begin{itemize}
		\item A class is said to be tightly encapsulated if and only if each variable declared is \textbf{private}.
		\item Eg:
		\codeblock{
			class A \{ \\
				\s private int a;  //Tightly encapsulated   \\ 
			\} \\
			class B extends A \{ \\
				\s int b; // Not tightly encapsulated \\
			\} 
		}
	
	\end{itemize}	
	
\end{flushleft}
