		\multicolumn{2}{l}{2. For loop and if condition in list}  \\
		\multicolumn{2}{l}{Syntax: \textbf{newlist=[x for x in list if condition]}}  \\ 
		
		\multicolumn{2}{l}{}  \\
		\multicolumn{2}{l}{Eg: Filter all even numbers in a list}  \\ \hline
		\multicolumn{1}{|l|}{Sample Code}  & \multicolumn{1}{l|}{Output}   
		\\ \hline
		\multicolumn{1}{|l|}{\begin{tabular}[c]{@{}l@{}}l1=[34,23,10,89] \\
				ans=[x for x in l1 if x\%2==0] \\
				print(ans)\end{tabular}}   &
		\multicolumn{1}{|l|}{[34, 10]}  \\
		\hline
		
		\multicolumn{2}{l}{}  \\
		\multicolumn{2}{l}{Eg: Filter all member with title case in a list}  \\ \hline
		\multicolumn{1}{|l|}{Sample Code}  & \multicolumn{1}{l|}{Output}   
		\\ \hline
		\multicolumn{1}{|l|}{\begin{tabular}[c]{@{}l@{}}l1=["Ravi","test","Raman"] \\
				ans=[x for x in l1 if x.istitle()] \\
				print(ans)\end{tabular}}   &
		\multicolumn{1}{|l|}{['Ravi', 'Raman']}  \\
		\hline	
		
		\multicolumn{2}{l}{}  \\
		\multicolumn{2}{l}{Eg: Append ".mp3" to all list member}  \\ \hline
		\multicolumn{1}{|l|}{Sample Code}  & \multicolumn{1}{l|}{Output}   
		\\ \hline
		\multicolumn{1}{|l|}{\begin{tabular}[c]{@{}l@{}}l1=["one.jpg","two.mp3","three.mp3 \\
				ans=[x for x in l1 if 'mp3' in x] \\
				print(ans)\end{tabular}}   &
		\multicolumn{1}{|l|}{['two.mp3', 'three.mp3']}  \\
		\hline
